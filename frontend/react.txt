# React project 1
# 1. Create a new React + Vite app
npm create vite@latest my-react-app -- --template react   # JS[web:7][web:9]
# or for TypeScript:
# npm create vite@latest my-react-app -- --template react-ts

# 2. Go into the project
cd my-react-app

# 3. Install dependencies
npm install

# 4. Start dev server
npm run dev
